基本操作與設定
<-
賦值運算子，將右側的值儲存到左側的物件名稱中
x <- 5 net_income <- income - taxes
rm()
移除物件。rm(list = ls()) 清空整個環境中所有物件
rm(x) # 移除單一物件 rm(list = ls()) # 清空全部
getwd() / setwd()
取得或設定目前的工作路徑（Working Directory），影響讀取/儲存檔案的預設位置
getwd() setwd("/Users/joe/Desktop")
c()
建立向量（vector），將多個值串接在一起
age <- c(28, 48, 47, 71) c("apple", "banana")
: 冒號運算子
快速建立整數序列，起始值到結束值（含兩端）
1:5 # 得到 1 2 3 4 5 5:1 # 得到 5 4 3 2 1
seq()
建立自訂的等差數列，可以設定間隔（by）或總長度（length.out）
seq(0, 1, by = 0.1) # 0 0.1 ... 1 seq(0, 3.99, by = 0.01)
matrix()
建立矩陣，可指定列數（nrow）、欄數（ncol），以及排列方向（byrow）
matrix(1:6, nrow = 2, ncol = 3) matrix(ncol = 5, nrow = 1000)
data.frame()
建立資料框架，不同欄位可以是不同資料種類（數值、字元等）
df <- data.frame(age, purchase) df2 <- data.frame(x = 1:3, y = c("a","b","c"))
cbind() / rbind()
將多個向量或資料框架依欄（column）或列（row）方向合併
cbind(Freq = t1, Percent = p1) # 欄合併 rbind(df1, df2) # 列合併
set.seed()
設定亂數種子，確保每次執行隨機運算時結果可重現
set.seed(1224) X <- sample(1:100, 30, TRUE)
? / help()
查詢函數的說明文件，在 RStudio 的 Help 分頁中顯示
?mean help(ggplot)
sprintf()
格式化字串輸出，類似 C 的 printf。常用於設定數字的小數位數格式
sprintf("%.1f", 3.1416) # "3.1" sprintf("R² = %.2f", 0.856)

library(ggplot2) library(haven)
if(!require()) {install...}
在 R Markdown 中安全安裝套件的寫法：若套件未安裝則安裝，否則跳過
if(!require("ggplot2")) { install.packages("ggplot2") } library(ggplot2)
p_load() pacman

library(pacman) p_load(haven, dplyr, psych, ggplot2)
data()
將 R 或已載入套件中的內建資料集載入到工作環境
data(msleep) data("penguins")
available.packages()

nrow(available.packages())

head(msleep) head(msleep, 2) # 只看前兩列 tail(df, 10)
dim() / nrow() / ncol()
回傳資料框架的「列數×欄數」；或分別取得列數、欄數
dim(msleep) # c(83, 11) nrow(msleep) # 83 ncol(msleep) # 11
str()
顯示資料框架的結構：每個變項的名稱、資料種類（chr/num/int/Factor）及前幾個值
str(msleep)
names() / row.names()
取得資料框架的欄位名稱向量（names）或列名稱（row.names）
names(airquality) row.names(msleep)
class()
回傳物件的資料種類，例如 "numeric"、"character"、"data.frame"、"factor" 等
class(x) # "numeric" class(msleep) # "tbl_df" class(penguins$sex) # "factor"

is.na() / sum(is.na())

is.na(msleep$sleep_rem) sum(is.na(msleep$sleep_rem)) # NA個數

library(haven) data <- read_sav("tscs2023q2.sav")

write_sav(rel, "rel.sav")

描述統計函數
mean()
計算平均數。若資料含 NA，需加 na.rm = TRUE 才能得到結果
mean(msleep$sleep_total) mean(msleep$sleep_rem, na.rm = TRUE)
median()
計算中位數（第 50 百分位數）。同樣支援 na.rm
median(msleep$sleep_total) median(data$age, na.rm = TRUE)
sd() / var()
計算標準差（sd）或變異量（var），分母為 n-1（樣本統計量）
sd(msleep$sleep_total) var(msleep$sleep_total)
max() / min()
取得向量中的最大值或最小值
max(airquality$Temp) # 97 min(msleep$sleep_total)
sum()
計算向量元素的總和。搭配邏輯判斷可計算符合條件的個數
sum(msleep$sleep_total > 8) # 計次 sum(is.na(msleep$sleep_rem)) # NA個數
quantile()
計算分位數，預設輸出 0%, 25%, 50%, 75%, 100%
quantile(msleep$sleep_total) quantile(data$age, 0.9, na.rm=T)
table()
建立次數分配表。加第二個變項則建立交叉表。useNA="ifany" 可顯示遺漏值
table(msleep$vore) table(msleep$vore, msleep$conservation) table(data$sex, useNA = "ifany")
proportions() / prop.table()
計算 table 物件各格的比例。margin=1 依列、margin=2 依欄計算
proportions(table(msleep$vore)) prop.table(table(data$a1)) proportions(table(x, y), margin = 1)
addmargins()
在 table 或交叉表的邊緣加上加總列/欄（Sum）
addmargins(round(prop.table(tbl), 4))
round()
四捨五入至指定小數位數
round(3.14159, 2) # 3.14 round(proportions(t1) * 100, 2)
aggregate()
依分組計算統計值，語法為 y ~ 分組變項，第三個參數為統計函數
aggregate(sleep_total ~ vore, data = msleep, mean)
cor() / cov()
計算相關係數或共變量。可對整個資料框架計算相關矩陣。use 參數處理 NA
cor(data$age, data$scale, use = "pairwise.complete.obs") cor(economics[, c(2,4:6)])

資料存取與篩選
$ 欄位存取
用 $ 符號存取資料框架中特定欄位（變項），回傳一個向量
msleep$sleep_total mean(airquality$Temp)
[ ] 索引
用 [列, 欄] 存取資料框架的特定位置。省略一方則取得整列或整欄
airquality[5, 4] # 第5列第4欄 airquality[5, ] # 第5列全部 airquality[, c(2,4,6)] # 第2,4,6欄 airquality[, -2] # 排除第2欄
which() / which.max()
which() 回傳符合條件的元素索引；which.max() 回傳最大值的索引位置
which(airquality$Temp > 90) which.max(msleep$sleep_total) msleep[which.max(msleep$sleep_total), ]
邏輯篩選
用邏輯條件篩選資料列；& 為且，| 為或，%in% 為「是否包含於」
airquality[airquality$Temp > 90, ] msleep[msleep$vore == "carni", ] data[data$a8 %in% c(1,2,3), ] msleep$sleep_total > 8 & msleep$bodywt > 50
select() dplyr
選取資料框架中特定欄位，可用欄位名稱或範圍（如 d15a:d15j）
sub_data <- select(.data = data, d15_an, d15_b, d15_c)

資料清理與轉換
mutate() dplyr
新增或修改資料框架中的欄位，可一次處理多個變項，常與 ifelse、case_when 搭配
data <- mutate(.data = data, birth = ifelse(a2y >= 0, a2y, NA), age = 112 - birth + 1, male = ifelse(a1 == 1, 1, 0))
ifelse()
條件判斷：若條件為 TRUE 回傳第一個值，否則回傳第二個值。可嵌套使用
ifelse(a2y >= 0, a2y, NA) ifelse(a1 == 1, 1, 0) ifelse(income < 0, NA, income)
case_when() dplyr
多條件對應重新編碼，適用於複雜的分類。語法為 條件 ~ 新值
edu <- case_when( b1 %in% c(1:5) ~ 1, # 國小以下 b1 %in% c(6:9) ~ 2, # 高中職 b1 %in% c(10:15) ~ 3, # 專科 b1 == 22 ~ NA)
recode() car
重新編碼，可直接指定新值對應舊值，支援轉換為因子變項（as.factor = T）
vore2 <- recode(msleep$vore, "'carni'='肉食性'; 'herbi'='草食性'", as.factor = T)
as.numeric() / as.character() / as.logical()
強制轉換資料種類。無法轉換時會產生 NA 及警告訊息
as.numeric(TRUE) # 1 as.character(3.14) # "3.14" as.numeric(data$age)
as.data.frame() / as.matrix()
在資料框架與矩陣之間轉換。tibble 轉換為 data.frame 可避免某些套件相容問題
penguins <- as.data.frame(penguins) mat <- as.matrix(airquality)
as.factor()
將向量轉換為因子變項（factor）
data$edu <- as.factor(data$edu) msleep$vore2 <- as.factor(msleep$vore)
is.numeric() / is.character() / is.logical()
檢查物件是否為指定的資料種類，回傳 TRUE 或 FALSE
is.numeric(3.14) # TRUE is.character("hello") # TRUE is.logical(FALSE) # TRUE
log()
計算自然對數（底數 e）；可指定 base 參數改變底數。常用於對數轉換
log(100) # 自然對數 log(100, base = 10) # 常用對數 l_income <- log(income + 1) # 避免log(0)
trunc()
截取數字的整數部分（無條件捨去小數）
temp <- trunc(j4b5 / 1000) trunc(3.99) # 3
t() 轉置
將矩陣或資料框架進行轉置（列變欄、欄變列）
relt <- t(rel) # 列變欄

因子變項（factor）
factor()
建立因子變項，可用 levels 指定類別及順序，labels 設定顯示名稱，ordered=TRUE 設定為有序（次序尺度）
smoke2 <- factor(smoke, levels = c("Never","Occasionally", "Regularly","Heavy"), ordered = TRUE) factor(sex, levels=c(2,1), labels=c("女","男"))
levels()

levels(smoke2) # 查看 levels(smoke2)[5] <- "Invalid" # 修改 levels(gender) <- c("雌性","雄性")
addNA()

smoke2 <- addNA(smoke2) table(smoke2) # NA 現在會出現
relevel()
更改因子變項的基準類別（reference level），用於迴歸分析中的虛擬變項設定
penguins$gender <- relevel( penguins$sex, ref = "male") msleep$vore2 <- relevel( as.factor(msleep$vore), ref = "herbi")

Base R 繪圖
plot()
通用繪圖函數；兩個向量產生散佈圖，模型物件產生診斷圖，pch 設定點的形狀
plot(msleep$sleep_total, msleep$sleep_rem) plot(msleep$sleep_total, msleep$sleep_rem, pch = 16) plot(TukeyHSD(m)) # Tukey 事後圖
hist()
繪製直方圖，用於呈現數值變項的分佈
hist(msleep$sleep_total)
boxplot()
繪製箱型圖，可用 y ~ 分組變項 語法繪製分組比較
boxplot(sleep_total ~ vore, data = msleep)
barplot()
繪製長條圖，通常以 table() 的結果作為輸入
barplot(table(msleep$vore))
abline()
在既有圖形上加入直線；v 為垂直線，h 為水平線
abline(v = 0) # 垂直線 abline(h = 0) # 水平線
polygon()
在圖形上繪製填色的多邊形，常用於標示常態分配的特定區域
polygon(x = c(2, xxx, 3), y = c(0, yyy, 0), col = "blue")
grid()
在圖形上加入背景格線
plot(x, y, pch = 16) grid()
par(mfrow=)
設定繪圖視窗的分割方式，c(2,2) 表示 2×2 的排列
par(mfrow = c(2, 2)) # 2x2 排列 plot(lm1) par(mfrow = c(1, 1)) # 恢復單圖
png() / pdf() / dev.off()
將圖形輸出儲存為圖檔：先呼叫 png/pdf 開啟檔案，繪圖後用 dev.off() 關閉並儲存

geom_point() geom_point(size = 2) geom_point(aes(colour = species))
geom_histogram()
直方圖。colour 設定輪廓色，fill 設定填色，binwidth 設定箱寬
geom_histogram() geom_histogram(colour = "cyan4", fill = "white", binwidth = 1)
geom_boxplot()
箱型圖，展示中位數、四分位距與離群值。width 調整箱子寬度
geom_boxplot() geom_boxplot(width = .1) geom_boxplot(aes(fill = species))
geom_violin()
小提琴圖，結合密度估計與箱型圖，呈現完整分佈形狀。alpha 調整透明度
geom_violin() geom_violin(alpha = .5)
geom_bar()
長條圖（計次）。fill 填色、position="dodge" 分組並排
geom_bar(fill = "cyan3", colour = "navy") geom_bar(position = "dodge") geom_bar(width = .8)

geom_text()

geom_segment()

geom_errorbar()

geom_vline() / geom_hline()

geom_vline(xintercept = 0, linetype = "dashed", color = "red")
geom_jitter()
將點加入隨機擾動避免重疊，常與箱型圖疊加使用，alpha 設定透明度

geom_qq() / geom_qq_line()

stat_function()
在圖上繪製數學函數的曲線，fun 指定函數，args 傳入參數

facet_wrap()

labs()
設定圖表標籤：title（標題）、x/y（軸標籤）、subtitle（副標題）
labs(x = "企鵝品種", y = "體重（公克）", title = "企鵝體重的箱型圖")

推論統計與假設檢定
t.test()
t 檢定。單樣本用 mu 設定假設值；雙樣本用 y ~ x 語法。alternative 設定單/雙尾
t.test(msleep$sleep_total, mu = 8) t.test(msleep$sleep_total, mu = 8, alternative = "greater") t.test(body_mass_g ~ sex, data = penguins, var.equal = TRUE)
var.test()
檢驗兩組變異量是否相等（F 檢定），在進行雙樣本 t 檢定前使用
var.test(msleep$sleep_total[msleep$vore=="carni"], msleep$sleep_total[msleep$vore=="herbi"]) var.test(body_mass_g ~ sex, data = penguins)
chisq.test()
卡方獨立性檢定（用於交叉表）或適合度檢定（設定 p 期望比例）
chisq.test(t1) # 獨立性 chisq.test(x = data$male, y = data$edu) chisq.test(table(data$male), p = c(0.46, 0.54)) # 適合度
fisher.test()
Fisher 精確檢定，適用於小樣本的 2×2 交叉表
fisher.test(TeaTasting, alternative = "greater")
shapiro.test()
Shapiro-Wilk 常態分配檢定。H₀：資料符合常態分配
shapiro.test(msleep$sleep_rem[ msleep$vore2 == "草食性"])
bartlett.test()
